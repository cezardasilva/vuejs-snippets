## Vue Router

### Router
  prefix: router
  body:
  ```javascript
  import Vue from 'vue'
  import VueRouter from 'vue-router'

  Vue.use(VueRouter)

  const routes = [
    {
      name: "YourRouteName",
      path: "/",
      component: YourComponent
    }
  ]

  export const router = new VueRouter({
    routes: routes,
    scrollBehavior (to, from, savedPosition) {
      return { x: 0, y: 0 }
    }
  })
  ```
